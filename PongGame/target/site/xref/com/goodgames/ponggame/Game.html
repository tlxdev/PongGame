74      private AI ai;
75  
76      /**
77       * Returns the player's bat.
78       *
79       * @return the player's bat
80       */
81      public Bat getPlayerBat() {
82          return player1;
83      }

85      public KeyboardInput getKeyboardInput() {
86          return keyboardInput;
87      }

99          return camera;

102     public Ball getBall() {
103         return ball;
104     }
105 
106     private Ball ball;
107 
108     /**
109      * Game instance.
110      */
111     public Game() {
112         camera = new Camera();
113         ball = new Ball(this);
114         player1 = new Bat(this);
115         player2 = new Bat(this);
116         wall1 = new Wall(this);
117         wall2 = new Wall(this);
118         wall1.move(2, 0);
119         wall2.move(-2, 0);
120         player2.move(0, -2);
121         player1.move(0, 3);
122 
123         ball.setDirection(new Vector3f(0.5f, 0, 0.5f).normalize());
124 
125         ai = new AI(Difficulty.EASY, this, player2);
126     }
127 
128     /**
129      * Renders the scene.
130      *
131      */
132     public void render() {
133 
134         //renderöi esineet
135         player1.render();
136         player2.render();
137         wall1.render();
138         wall2.render();
139         ball.render();
140 
141     }
142 
143     /**
144      * Updates the game logic and checks for input.
145      *
146      * @param deltaTime the amount of time passed between frames
147      */
148     public void update(double deltaTime) {
149 
150         ball.update(deltaTime);
151 
152         player2.update(deltaTime);
153         /*
154          Päivitä fps vain joka viidestoista frame(muuten se on liian vaikea lukea)     
155          */
156         if (ai.getDifficulty().equals(Difficulty.EASY)) {
157             ai.update(deltaTime);
158         } else if (frameCounter % 6 == 0) {
159             ai.update(deltaTime);
160         }
161         if (frameCounter == 15) {
162             frameCounter = 0;
163         }
164         if (frameCounter == 0) {
165             GLFW.glfwSetWindowTitle(windowId, "Pong FPS: " + (int) (1 / deltaTime));
166         }
167 
168         frameCounter++;
169 
170         input.update(deltaTime);
171         keyboardInput.update(deltaTime);
172 
173         if (ball.getY() > 3.2 || ball.getY() < -2) {
174             System.out.println("lose");
175             ball = new Ball(this);
176 
177             ball.setDirection(new Vector3f(Math.max(new Random().nextFloat(), 0.5f), 0, Math.max(new Random().nextFloat(), 0.5f)).normalize());
178             ai.setBall(ball);
179         }
180 
181     }
182 
183     /**
184      * Sets the window id, sets up input listeners for the window.
185      *
186      * @param id the id of the window
187      */
188     public void setWindowId(long id) {
189         this.windowId = id;
190         input = new MouseInput(id, this);
191         keyboardInput = new KeyboardInput(id, this);
192 
193     }
