116     /**
117      * Game instance.
118      */
119     public Game() {
120         camera = new Camera();
121         ball = new Ball(this);
122         gameObjects.add(ball);
123         player1 = new Bat(this);
124         player2 = new Bat(this);
125         wall1 = new Wall(this);
126         wall2 = new Wall(this);
127         wall1.move(2, 0);
128         wall2.move(-2, 0);
129         player2.move(0, -2);
130         player1.move(0, 3);
131 
132         ball.setDirection(new Vector3f(0.5f, 0, 0.5f).normalize());
133 
134         ai = new AI(Difficulty.EASY, this, player2);
135     }

159     /**
160      * Updates the game logic and checks for input.
161      *
162      * @param deltaTime the amount of time passed between frames
163      */
164     public void update(double deltaTime) {
165 
166         //ball.update(deltaTime);
167         for (GameObject go : gameObjects) {
168             go.update(deltaTime);
169             Ball b = (Ball) go;
170             if (b.needsToBeDeleted()) {
171                 toDelete.add(b);
172             }
173         }
174 
175         for (GameObject go : toDelete) {
176             gameObjects.remove(go);
177         }
178 
179         player2.update(deltaTime);
180         /*
181          Päivitä fps vain joka viidestoista frame(muuten se on liian vaikea lukea)     
182          */
183         if (ai.getDifficulty().equals(Difficulty.EASY)) {
184             ai.update(deltaTime);
185         } else if (frameCounter % 6 == 0) {
186             ai.update(deltaTime);
187         }
188         if (frameCounter == 15) {
189             frameCounter = 0;
190         }
191         if (frameCounter == 0) {
192             GLFW.glfwSetWindowTitle(windowId, "Pong FPS: " + (int) (1 / deltaTime) + " scores " + player1Score + ":" + player2Score);
193         }
194 
195         frameCounter++;
196 
197         input.update(deltaTime);
198         keyboardInput.update(deltaTime);
199 
200         if (ball.getY() > 3.2 || ball.getY() < -2) {
201             if (ball.getY() > 3.2) {
202                 player2Score++;
203             } else {
204                 player1Score++;
205             }
206             System.out.println("lose");
207             ball.setTimeToLive(0.5f);
208             ball = new Ball(this);
209             gameObjects.add(ball);
210             ball.setDirection(new Vector3f(Math.max(new Random().nextFloat(), 0.5f), 0, Math.max(new Random().nextFloat(), 0.5f)).normalize());
211             ai.setBall(ball);
212         }
213 
214     }
215 
216     public void addBall() {
217 
218         Ball newBall = new Ball(this);
219         gameObjects.add(newBall);
220         newBall.setDirection(new Vector3f(Math.max(new Random().nextFloat(), 0.5f), 0, Math.max(new Random().nextFloat(), 0.5f)).normalize());
221 
222     }
